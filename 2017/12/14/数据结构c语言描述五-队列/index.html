数据结构c语言描述五-队列

发表于 2017-12-14 | 分类于 数据结构

和栈类似，队列也是一种辅助性质的数据结构，队列是只允许在一端进行插入，在另一端进行删除操作的线性表，只允许插入的一端称为对头，只允许删除的一端称为队头

和栈类似，队列也有两种方式实现，一种以顺序表方式实现，一般为循环队列，一种以链表方式实现

队列的定义

顺序表方式实现

typedef struct _Noe{
    int data;
}Node;

typedef struct Queue{
    Node* base;
    int front;
    int rear;
}Queue;

链表方式实现

typedef struct _Node{
    int data;
    struct _Node* next;
}Node;

typedef struct Queue{
    Node* head;
    Node* last;
    int length;
}Queue;

基本操作

Queue InitQueue();
int QueueClear(Queue Q);
int QueueIsEmpty(Queue Q);
int QueueLength(Queue Q);
Node GetHead(Queue Q);

int EnQueue(Queue Q,Node n);

队列的初始化

顺序表方式实现

Queue* Q = (Queue*)malloc(sizeof(Queue));
if(Q == NULL) return NULL;
Q->base = (Node*)malloc(sizeof(Node) * SIZE);
if(Q->base == NULL) return NULL;
Q->front = 0;
Q->rear = 0;
return Q;

链表方式实现

Queue* Q = (Queue*)malloc(sizeof(Queue));
  if(Q == NULL) return NULL;
  Q->head = (Node*)malloc(sizeof(Node));
  if(Q->head == NULL)return NULL;
  Q->length = 0;
  Q->head->next = NULL;
  Q->last = Q->head;
  return Q;

顺序表入队

if(QueueIsDestroy(Q) == 1) return 0;

出队操作

顺序表出队

if(QueueIsDestroy(Q) == 1) return NULL;
  if(QueueIsEmpty(Q) == 1) return NULL;
  Node* n = &Q->base[Q->front];
  Q->front = (++Q->front) % SIZE;
  return n;

链表出队

if(Q->head == NULL || Q->head->next == NULL) return NULL;
