数据结构c语言描述三-链表

发表于 2017-12-12 | 分类于 数据结构 |

链表是一种这样的数据结构，其中的各对象按线性顺序排列，然而与数组不同的是，链表的顺序是由各个对象里的指针决定的

链表有很多构造方法:单向链表，单向循环链表，双向链表，双向循环链表等，这里以双向非循环链表为例

双向链表的基本操作

List ListInit(); //初始化链表

int ListLength(List L); //返回链表的长度

int LocaleList(List L,Node n); //返回链表中指定元素第一次出现的下标索引，没有返回-1

Node ListGet(List L,int index); //返回链表指定索引位置的节点元素

int ListClear(List L); //清空链表，成功返回1,失败返回0

int ListEmpty(List L); //判断链表是否为空，是空返回1,不是返回0

int ListInsert(List L,int index,Node n); //在链表指定位置插入指定元素，插入成功返回1,失败返回0

int ListDelete(List L,int index); //删除链表表指定索引位置的元素，删除成功返回1,失败返回0

void printList(List L); //打印链表

双向链表的定义

typedef struct _Node{ 

int data;                 //数据类型，可以多个
struct _Node* next;       //后继指针
struct _Node* prior;      //前驱指针

}Node;

typedef struct LinkedList{

Node* head;       //这里构建的是一个带有头指针的链表，这个头指针不保存数据
Node* last;       //存有一个指向链表末尾的指针，因为链表只有顺序查找，尾指针在某些时候可以提高效率
int length;       //保存链表中数据接节点的个数

}LinkedList;

双向链表的初始化

LinkedList* ListInit(){

LinkedList* L = (LinkedList*)malloc(sizeof(LinkedList));
if(L == NULL)
    return NULL;
  L->head = (Node*)malloc(sizeof(Node));
  L->head->next = NULL;
  L->head->prior = NULL;
  L->last = L->head;
  L->length = 0;
  return L;

}

双向链表的插入

int ListInsert(LinkedList L,int i,Node n){

if(i < 0){//判断元素要插入的位置是否符合规范
    return 0;
}
if(i > L->length){
    //如果要插入的位置比链表的长度还要大，那么插在尾部
    n->prior = L->last;
    n->next = NULL;
    L->last->next = n;
    L->last = n;
}else{
//如果插入的位置在以存在链表的中
    int index = 0;
    Node* ptr = L->head;
//先找到要插入的位置，这个位置是保存在ptr指针中
    while(index != i){
        index++;        
        ptr = ptr->next;
    }

    n->next = ptr;
    Node* prior = ptr->prior;
    n->prior = prior;
    prior->next = n;
    ptr->prior = n;
}
L->length++;
return 1;

}

双向链表的删除

int ListDelete(LinkedList* L,int index){

if(index < 0) return 0;//判断要删除的位置的索引是否符合规范
if(ListEmpty(L) == 1) return 0;  //判断链表是否为空
Node* ptr = L->head->next;
if(index == 1){//删除头指针
    L->head->next = ptr->next;
    ptr->prior = L->head;
    free(ptr);
    return 1;
}else if(index >= L->length){//删除位置大于链表长度，删除尾指针
    ptr = L->last;
    L->last = ptr->prior;
    L->last->next = NULL;
    free(ptr);
    return 1;
}else{//删除位置在中间
    int i = 1;
    while(i != index){
        i++;
        ptr = ptr->next;
    }
    Node* temp = ptr->prior;
    temp->next = ptr->next;
    ptr->next->prior = temp;
    free(ptr);
    return 1;
}

}
