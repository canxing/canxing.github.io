数据结构c语言描述二-线性表

发表于 2017-12-11 | 分类于 数据结构 |

线性表是一种线性结构，在c语言中，表示为在堆中分配一段连续的存储空间来表示

线性表的基本操作

List ListInit(); //初始化线性表

int ListLength(List L); //返回线性表的长度

int LocaleList(List L,Node n); //返回线性表中指定元素第一次出现的下标索引，没有返回-1

Node ListGet(List L,int index); //返回线性表指定索引位置的节点元素

int ListClear(List L); //清空线性表，成功返回1,失败返回0

int ListEmpty(List L); //判断线性表是否为空，是空返回1,不是返回0

int ListInsert(List L,int index,Node n); //在线性表指定位置插入指定元素，插入成功返回1,失败返回0

int ListDelete(List L,int index); //删除线性表指定索引位置的元素，删除成功返回1,失败返回0

void printList(List L); //打印线性表

线性表的类型定义

typedef struct Node{

int data;//节点元素类型，可以自定义元素

}Node;

typedef struct linearList{

Node* elem;
int length;//保存该线性表已经存储的个数
int size;//保存这个线性表可以存储的最大个数

}LinearList;

插入数据

int ListInsert(LinearList* L,int i,Node n){

if(i < 0)//判断插入位置是否符合规范
    return 0;
if(L->length >= L->size){//如果线性表中的数据已经大于或等于线性表的最大存储数，返回，还可以使用还可以使用relloc函数动态增加数量
    return 0;
}
int length = L->length;
if( i <= length){//如果插入位置是在已经存在的线性表中，把索引位置以及之后的元素依次先后移动一位
    for(;length != i - 1;length--){
        L->elem[length] = L->elem[length - 1];
    }
}
//如果插入位置是在已经插入数据长度之外，那么直接插入数据，
//如果不是，那么length索引的元素变为插入数据
L->elem[length] = n;
L->length++;
return 1;

}

删除数据

int ListDelete(LinearList* L,int index){

if(index < 0 || index >= L->length)//判断删除索引的位置是否在范围外
    return 0;
for(int i = index;i < L->length; i++){//将删除元素之后的所有元素依次向后移动一位
    L->elem[i] = L->elem[i + 1];
}
L->length--;
return 1;

}
